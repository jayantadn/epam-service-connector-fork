# ---------------------------------------------------------------------------
# Host-side Python dependencies for the EV Range Extender lab.
#
# These two libraries are the ONLY pip packages the developer needs on
# their own machine. Everything that runs *inside* the VMs (kuksa-client,
# eclipse-zenoh on VM1/VM2) is installed automatically by cloud-init on
# first boot, so it does not appear here.
#
# Install (one of the following):
#
#   # 1) Inside an active virtualenv  (recommended)
#   python3 -m pip install -r requirements.txt
#
#   # 2) System-wide Python on Debian / Ubuntu (no venv)
#   python3 -m pip install --user -r requirements.txt
#
#   # 3) System Python that refuses pip without a flag (PEP 668)
#   python3 -m pip install --user --break-system-packages -r requirements.txt
#
# Note: setup.py also runs `sudo apt install python3-yaml`. That apt
# package only covers the *system* python3. If you are running setup.py
# from a virtualenv, install PyYAML via this requirements.txt as well.
# ---------------------------------------------------------------------------

# Required by tools/compose_userdata.py (called by setup.py) to render
# the cloud-init user-data templates that are seeded into VM1 and VM2.
PyYAML>=6.0

# Required by hardware-sim/pytk_dashboard.py - the Tk dashboard you run
# on the host to drive Battery / HVAC / Seat signals over Zenoh into
# the ECU services on VM1 and VM2.
eclipse-zenoh>=1.0.0
